\section{Files \& Directories}
\begin{tabular}{ll}
    \tt{cat a}              & print a                               \\
    \tt{cd dir/}            & go to dir/                            \\
    \tt{cd ..}              & go to parent dir                      \\
    \tt{pwd}                & print current dir                     \\
    \tt{cp a b}             & copy a to b                           \\
    \tt{cp a dir/}          & copy a into dir/                      \\
    \tt{cp -r sub/ exist-dir/}    & copy sub/ into exist-dir/       \\
    \tt{cp -r sub/ new-dir/} & copy sub/ as new-dir/                \\
    \tt{mv a b}             & rename a to b                         \\
    \tt{mv a dir/}          & move a into dir/                      \\
    \tt{mv sub/ exist-dir/}       & move sub/ into exist-dir/       \\
    \tt{mv sub/ new-dir/}   & rename sub/ as new-dir/               \\
    \tt{rm a}               & remove a                              \\
    \tt{rm -r dir/}         & remove dir/                           \\
    \tt{ls}                 & list files under wd                   \\
    \tt{ls -a}              & list files, incl. hidden              \\
    \tt{ls -l}              & list files, show details              \\
    \tt{ls dir/}            & list files under dir/                 \\
    \tt{mkdir dir/}         & create dir/                           \\
    \tt{mkdir -p dir/sub/}  & create dir/sub/ directly
\end{tabular}

\section{Utilities}
\begin{tabular}{ll}
    \tt{head -n 5 a}        & print first 5 lines of a              \\
    \tt{grep pattern files} & grep for pattern in files             \\
    \tt{grep -i}            & ignore case                           \\
    \tt{grep -v}            & invert match                          \\
    \tt{grep -o}            & only show matched part                \\
    \tt{grep -E}            & extended regex                        \\
    \tt{sed -E 's/a/b/'}    & replace first 'a' with 'b'            \\
    \tt{sed -E 's/a/b/g'}   & replace all 'a's with 'b' (global)    \\
    \tt{sort file}          & sort lines in file                    \\
    \tt{sort -f}            & ignore case                           \\
    \tt{sort -n}            & numerical order                       \\
    \tt{sort -r}            & reverse                               \\
    \tt{uniq}               & remove duplicate lines                \\
    \tt{comm -12 a b}       & print lines in both a and b           \\
    \tt{diff a b}           & difference between a and b
\end{tabular}

\section{Ctrl+}
\begin{tabular}{ll}
    \tt{A} & move cursor to the beginning                          \\
    \tt{E} & move cursor the the end                               \\
    \tt{W} & delete one word to the left                           \\
    \tt{U} & delete everything to the left                         \\
    \tt{K} & delete everything to the right                        \\
    \tt{C} & abort                                                 \\
    \tt{R} & search history                                        \\
    \tt{L} & clear screen
\end{tabular}

\section{Regex (ERE)}
\begin{tabular}{ll}
    \verb|.|            & any character                             \\
    \verb|\.|           & literal dot                               \\
    \verb|[aeiou]|      & any vowel                                 \\
    \verb|[^aeiou]|     & anything but a vowel                      \\
    \verb|[0-9]|        & any digit                                 \\
    \verb|[A-Za-z]|     & any letter                                \\
    \verb|^|            & beginning of line/string                  \\
    \verb|$|            & end of line/string                        \\
    \verb|A?|           & 0 or 1 A                                  \\
    \verb|A+|           & $\ge 1$ A's                               \\
    \verb|A*|           & $\ge 0$ A's                               \\
    \verb|A{6}|         & 6 A's                                     \\
    \verb|A{,6}|        & 0--6 A's                                  \\
    \verb|A{4,6}|       & 4--6 A's                                  \\
    \verb|A{4,}|        & $\ge 4$ A's                               \\
    \verb!(ls|cd|rm)!   & one of ls, cd, and rm                     \\
    \verb|(...)|        & capturing group                           \\
    \verb|\1, \2|       & backreference to group 1, 2               \\
    \verb|\w|           & letters, digits, \_                       
\end{tabular}

\section{Permissions \& Ownership}
\begin{tabular}{ll}
    \tt{r, w, x}            & read (4), write (2), execute (1)      \\
    \tt{chmod 755 a}        & \tt{rwx} owner, \tt{r-x} group/others \\
    \tt{chmod +x a}         & add execute permission for all        \\
    \tt{chown u:g a}        & change owner to \tt{u}, group to \tt{g}\\
    \tt{chown -R u:g dir/}  & change owner/group recursively        
\end{tabular}

\section{Process \& System}
\begin{tabular}{ll}
    \tt{ps aux}             & list all running processes            \\
    \tt{pgrep name}         & find process ID (PID) by name         \\
    \tt{kill PID}           & terminate process by PID              \\
    \tt{pkill name}         & terminate process by name             \\
    \tt{pkill -9 name}      & forcefully terminate                  \\
    \tt{sudo reboot}        & restart system immediately            \\
    \tt{sudo shutdown now}  & shut down immediately                 \\
    \tt{sudo shutdown +10}  & shut down in 10 minutes               \\
    \tt{sudo shutdown -c}   & cancel a pending shutdown             
\end{tabular}

\section{Environment}
\begin{tabular}{ll}
    \verb|~/.bashrc|        & script run on interactive startup     \\
    \tt{alias c='clear'}    & create a custom shortcut              \\
    \verb|export PATH=...|  & add paths to executable locations
\end{tabular}

\section{Scripting}
\begin{tabular}{ll}
    \verb|#!/bin/bash|  & shebang (specify interpreter)             \\
    \tt{var=val}        & define variable (no spaces!)              \\
    \verb|$var|         & access variable                           \\
    \verb|export V=val| & expose variable to child procs            \\
    \verb|$()|          & output of command substitution            \\
    \tt{""}             & double quote (allows expansion)           \\
    \tt{''}             & single quote (literal string)             \\
    \tt{./script.sh}    & execute in subshell (needs \tt{+x})       \\
    \tt{source script.sh}& execute in current shell                 \\
    \verb|c1 && c2|     & run c2 if c1 succeeds (exit 0)            \\
    \verb!c1 || c2!    & run c2 if c1 fails (exit $\neq 0$)        \\
    \verb|! cmd|        & negate exit code                          
\end{tabular}
